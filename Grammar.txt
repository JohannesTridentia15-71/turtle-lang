LINE = QUERY [save to GRAPH_NAME] | evaluate OPERATION [save to FILE_NAME]

QUERY = COMBINE_QUERY | ADD_QUERY | REPLACE_QUERY | CONSTRUCT_QUERY | DELETE_QUERY
SELECT_QUERY = select from GRAPH_NAME | select from GRAPH_NAME where FILTER_START | SELECT_ELEMENT_QUERY
SELECT_QUERY_NO_ELEMENT = select from GRAPH_NAME | select from GRAPH_NAME where FILTER_START
SELECT_ELEMENT_QUERY = select ELEMENT from GRAPH_NAME where FILTER_START | LITERAL
SELECT_OBJECT_QUERY = select #object from GRAPH_NAME where FILTER_START
COMBINE_QUERY = combine COMBINE_QUERY SELECT_QUERY | SELECT_QUERY
ADD_QUERY = add BRANCH to GRAPH_NAME
REPLACE_QUERY = in GRAPH_NAME replace SELECT_QUERY_NO_ELEMENT with OBJECT | in GRAPH_NAME replace SELECT_QUERY_NO_ELEMENT with SELECT_OBJECT_QUERY
CONSTRUCT_QUERY = construct GRAPH_NAME
DELETE_QUERY = delete GRAPH_NAME | delete GRAPH_NAME where FILTER_START

OPERATION = ARITHMETIC_OPERATION | LOGICAL_OPERATION | GRAPH_OPERATION
ARITHMETIC_OPERATION =  [+-*/] (SELECT_ELEMENT_QUERY) (SELECT_ELEMENT_QUERY)
==> elements must be floats/reals/integers (type checking performed upon execution)
==> returns a float/real/integer  

LOGICAL_OPERATION = [< > == != <= >=] (SELECT_ELEMENT_QUERY) (SELECT_ELEMENT_QUERY)
==> elements must be of the same type (type checking performed upon execution)
==> returns a boolean

GRAPH_OPERATION = [union intersection difference] (SELECT_ELEMENT_QUERY) (SELECT_ELEMENT_QUERY) | GRAPH_OPERATION_SINGLE

GRAPH_OPERATION_SINGLE = [min max] (SELECT_ELEMENT_QUERY)
==> takes the min/max object value for each subject/predicate pair 
==> requires all objects to be of the same type and either ints or strs 

GRAPH_NAME = [a-zA-Z0-9].ttl
ELEMENT = #subject | #predicate | #object
BRANCH = <SUBJECT> <PREDICATE> <OBJECT>
SUBJECT = URI
PREDICATE = URI
OBJECT = URI | LITERAL
==> Note URI grammar comes from RFC 3986 


FILTER_START = FILTER and FILTER_FINAL | FILTER or FILTER_FINAL | not FILTER
FILTER = FILTER and FILTER_FINAL | FILTER or FILTER_FINAL | not FILTER | LOGICAL_OPERATION
FILTER_FINAL = LOGICAL_OPERATION
